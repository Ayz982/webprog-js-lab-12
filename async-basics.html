Завдання 1. Основи async/await

Fetch Number

Консольний варіант виконання


function fetchNumber() {
    return new Promise(resolve => {
        setTimeout(() => {
            resolve(9);
        }, 2000);
    });
}

async function main() {
    try {
        let number = await fetchNumber();
        console.log('Отримане число:', number);
    } catch (error) {
        console.error('Помилка:', error.message);
    }
}
